6) Fai la somma di tutti i prezzi delle prenotazioni per le stanze del primo piano

SELECT SUM(`price`) AS `Somma Prenotazioni`,stanze.floor
FROM `prenotazioni`
INNER JOIN `stanze`
ON stanze.id = prenotazioni.stanza_id
INNER JOIN `pagamenti`
ON pagamenti.prenotazione_id = prenotazioni.id
WHERE stanze.floor = 1

------------------------------------------------------------------------------------------

7) Prendi i dati di fatturazione per la prenotazione con id=7

SELECT paganti.name, paganti.lastname, paganti.address
FROM `paganti`
INNER JOIN `pagamenti`
ON pagamenti.pagante_id = paganti.id
INNER JOIN `prenotazioni`
ON prenotazioni.id = pagamenti.prenotazione_id
WHERE prenotazioni.id = 7;
